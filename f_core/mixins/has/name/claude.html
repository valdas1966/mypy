What is HasName?

A mixin that provides name-based identification with full comparison and string representation support. Inherits from Comparable and Printable.

Identification

name property (get/set)

Comparison

Sort by name alphabetically

Equality

Equal if names match

Hashable

Use in sets and dicts

Quick Start

from f_core.mixins.has.name import HasName

obj = HasName(name="MyObject")

# Properties
print(obj.name)    # "MyObject"

# String representation
print(str(obj))    # "MyObject"
print(repr(obj))   # <HasName: Name=MyObject>

# Comparison (from Comparable)
obj2 = HasName(name="Another")
obj2 < obj  # True ("Another" < "MyObject")

# Collections (from Equable)
{obj, obj2}  # Works in sets

def key_comparison(self) -> str:
    return self._name or ''  # None → empty string

Basic Usage

from f_core.mixins.has.name import HasName

# Create with name
obj = HasName(name="MyObject")
print(obj.name)    # "MyObject"
print(str(obj))    # "MyObject"
print(repr(obj))   # <HasName: Name=MyObject>

# Mutate name
obj.name = "NewName"
print(obj.name)    # "NewName"

Sorting

items = [
    HasName("zebra"),
    HasName("apple"),
    HasName(None),
    HasName("banana")
]

# Sort alphabetically
sorted(items)  # [None, apple, banana, zebra]

# Min/max
min(items)  # HasName(None)
max(items)  # HasName("zebra")

# Reverse sort
sorted(items, reverse=True)  # [zebra, banana, apple, None]

Collections

# Sets work (via Equable)
names = {HasName("alice"), HasName("bob"), HasName("alice")}
len(names)  # 2 (duplicate removed)

HasName("alice") in names  # True

# Dict keys work
data = {
    HasName("config"): {"debug": True},
    HasName("data"): [1, 2, 3]
}
data[HasName("config")]  # {"debug": True}

Inheritance

class Person(HasName):
    def __init__(self, name: str, age: int):
        super().__init__(name=name)
        self.age = age

    def __repr__(self):
        return f"<Person: {self.name}, age={self.age}>"

# All HasName capabilities inherited
people = [Person("Bob", 30), Person("Alice", 25)]
sorted(people)  # [Alice, Bob] - sorted by name!

# Equality works
Person("Bob", 30) == Person("Bob", 25)  # True (same name)

Factory Usage

# Create test instances
a = HasName.Factory.a()       # name='A'
b = HasName.Factory.empty()   # name=''
c = HasName.Factory.none()    # name=None

# Compare
a > b   # True ('A' > '')
b == c  # True (both key to '')

# Sort
sorted([a, b, c])  # [empty, none, a]
