A. Purpose

  1. Collection of 6 mixin classes that inject common object properties via multiple inheritance.
  2. Three categories: identity (key, name), structure (children, parent), 2D positioning (row_col, rows_cols).
  3. Each mixin is independently composable — mix any combination into a single class.

G. Usage Examples

Identity via Key 
from f_core.mixins.has.key import HasKey

a = HasKey[str](key='A')
b = HasKey[str](key='B')

print(a.key)       # 'A'
print(str(a))      # 'A'
print(repr(a))     # '<HasKey: Key=A>'
print(a < b)       # True
print(a == HasKey[str](key='A'))  # True
print(hash(a))     # hash('A')
Name Property 
from f_core.mixins.has.name import HasName

obj = HasName(name='MyObject')
print(str(obj))    # 'MyObject'
print(repr(obj))   # '<HasName: Name=MyObject>'

empty = HasName()
print(str(empty))  # 'NoName'
Grid Positioning (HasRowCol) 
from f_core.mixins.has.row_col import HasRowCol

cell = HasRowCol(row=1, col=2)
print(cell)                     # (1,2)
print(cell.neighbors())        # [(0,2), (1,3), (2,2), (1,1)]
print(cell.distance(HasRowCol(3, 4)))  # 4

# Comparison and hashing
a = HasRowCol(0, 0)
b = HasRowCol(1, 1)
print(a < b)                   # True
print({a, b, HasRowCol(0, 0)}) # {(0,0), (1,1)}
Shape Dimensions (HasRowsCols) 
from f_core.mixins.has.rows_cols import HasRowsCols

grid = HasRowsCols(rows=3, cols=4)
print(grid.shape())             # '(3,4)'
print(len(grid))                # 12
print(grid.is_within(2, 3))     # True
print(grid.is_within(3, 4))     # False

square = HasRowsCols(rows=5)    # cols defaults to 5
print(square.shape())           # '(5,5)'
Parent-Child Trees 
from f_core.mixins.has.parent import HasParent
from f_core.mixins.has.children import HasChildren

# Parent chain
a = HasParent.Factory.a()       # ParentWithName(name='A')
b = HasParent.Factory.b()       # ParentWithName(name='B', parent=a)
print(b.parent.name)            # 'A'
print(len(b.path_from_root()))  # 2  (a -> b)

# Children list
parent = HasChildren()
parent.add_child(child=HasChildren())
parent.add_child(child=HasChildren())
print(len(parent.children))     # 2
Mixin Composition 
from f_core.mixins.has.name import HasName
from f_core.mixins.has.row_col import HasRowCol

class Cell(HasRowCol, HasName):
    def __init__(self, row: int, col: int, name: str) -> None:
        HasRowCol.__init__(self, row=row, col=col)
        HasName.__init__(self, name=name)

cell = Cell(row=1, col=2, name='Start')
print(cell.name)       # 'Start'
print(cell.row)        # 1
print(cell.key)        # (1, 2) — from HasRowCol
