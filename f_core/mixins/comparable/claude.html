What is Comparable?

Comparable extends Equable to add ordering operators. While Equable provides equality (==, !=), Comparable adds the ability to sort and compare objects (<, >, <=, >=).

Same key_comparison() Pattern

Like Equable, you only implement key_comparison(). All 6 operators are derived from it automatically.

Quick Start

from f_core.mixins.comparable import Comparable

class Score(Comparable):
    def __init__(self, points: int):
        self.points = points

    def key_comparison(self) -> int:
        return self.points

# All operators work automatically
s1 = Score(100)
s2 = Score(150)

s1 < s2    # True
s2 > s1    # True
s1 == Score(100)  # True (from Equable)

sorted([s2, s1])  # [Score(100), Score(150)]

How Comparison Works

→
a.__lt__(b)
→
a.key_comparison()
<
b.key_comparison()
def __lt__(self, other: Comparable) -> bool:
    return self.key_comparison() < other.key_comparison()

def __le__(self, other: Comparable) -> bool:
    return self.key_comparison() <= other.key_comparison()

# Same pattern for __gt__ and __ge__

Abstract Method

@abstractmethod
def key_comparison(self) -> Equable:
    """Return the key for comparison."""
    pass

Same as Equable.key_comparison() - one method drives all operators.

a = Comparable.Factory.a()  # Temp(key='A')
b = Comparable.Factory.b()  # Temp(key='B')
a < b  # True ('A' < 'B')

HasName Implementation

class HasName(Comparable, Printable):
    def __init__(self, name: str = None):
        self._name = name

    def key_comparison(self) -> str:
        return self._name or ''  # Empty sorts first

# Sorting works
names = [HasName("Zebra"), HasName("Apple"), HasName(None)]
sorted(names)  # [None, Apple, Zebra]

HasRowCol Implementation

class HasRowCol(Comparable, Printable):
    def __init__(self, row: int, col: int):
        self._row = row
        self._col = col

    def key_comparison(self) -> tuple[int, int]:
        return (self._row, self._col)

# Grid positions sort top-to-bottom, left-to-right
cells = [HasRowCol(1, 2), HasRowCol(0, 5), HasRowCol(1, 0)]
sorted(cells)  # [(0,5), (1,0), (1,2)]

Basic Implementation

from f_core.mixins.comparable import Comparable

class Score(Comparable):
    def __init__(self, points: int):
        self.points = points

    def key_comparison(self) -> int:
        return self.points

# Comparison operators
s1 = Score(100)
s2 = Score(150)

s1 < s2   # True
s2 > s1   # True
s1 <= s1  # True
s1 >= s1  # True
s1 == Score(100)  # True

Sorting & Min/Max

scores = [Score(50), Score(100), Score(75)]

sorted(scores)  # [Score(50), Score(75), Score(100)]
min(scores)     # Score(50)
max(scores)     # Score(100)

# Reverse sort
sorted(scores, reverse=True)  # [Score(100), Score(75), Score(50)]

Multi-Criteria Sorting

class Person(Comparable):
    def __init__(self, name: str, age: int):
        self.name = name
        self.age = age

    def key_comparison(self) -> tuple[int, str]:
        return (self.age, self.name)  # Sort by age, then name

people = [
    Person("Alice", 30),
    Person("Bob", 25),
    Person("Charlie", 30),
]

sorted(people)
# [Bob(25), Alice(30), Charlie(30)]
# Bob first (youngest), then Alice < Charlie (same age, alphabetical)

With Collections

# Sets work (via Equable's __hash__)
unique = {Score(100), Score(50), Score(100)}
len(unique)  # 2 (duplicates removed)

# Dict keys work
rankings = {
    Score(100): "Gold",
    Score(50): "Bronze"
}
rankings[Score(100)]  # "Gold"

Generic Wrapper

from typing import Generic, TypeVar

Key = TypeVar('Key')

class PriorityKey(Generic[Key], Comparable):
    def __init__(self, key: Key):
        self._key = key

    def key_comparison(self) -> Key:
        return self._key

# Works with any comparable type
PriorityKey(10) < PriorityKey(20)        # int
PriorityKey("a") < PriorityKey("b")      # str
PriorityKey((1,2)) < PriorityKey((1,3))  # tuple

Common Patterns

Pattern key_comparison() Effect
Simple value return self.val Sort by single attribute
Tuple return (a, b) Sort by a, then b
Handle None return val or '' None sorts first
Descending return -self.val Reverse order
Case-insensitive return name.lower() Ignore case
