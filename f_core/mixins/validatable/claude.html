What is Validatable?

A mixin that adds validation state to objects. The validation state is a boolean that can be checked using Python's truthiness: if obj:

Key Feature: __bool__()

Objects become truthy/falsy based on their validation state. Use if obj: instead of if obj.is_valid:

Validation States

✓
True
Valid / Success
if obj: ✓
✗
False
Invalid / Failure
if obj: ✗
?
None
Uninitialized
if obj: ✗

Two Classes

Validatable

Read-Only

  • Set at construction
  • Cannot change later
  • Use for fixed validity

ValidatablePublic

Mutable

  • Set at construction
  • Can change via methods
  • Use for dynamic validity

Quick Start

from f_core.mixins.validatable import Validatable

class Response(Validatable):
    def __init__(self, status: int, data: dict):
        is_valid = status == 200 and data is not None
        Validatable.__init__(self, is_valid=is_valid)
        self.data = data

response = Response(200, {"ok": True})

if response:  # Calls __bool__()
    process(response.data)

Common Patterns

cell.set_invalid()  # Mark as obstacle

Multi-Mixin

class Status(HasName, Printable, Validatable):
    def __init__(self, code: int):
        HasName.__init__(self, name=str(code))
        Validatable.__init__(self, is_valid=code == 200)

HTTP Response

from f_core.mixins.validatable import Validatable

class Response(Validatable):
    def __init__(self, status: int, data: dict):
        # Valid if 200 AND has data
        is_valid = status == 200 and data is not None
        Validatable.__init__(self, is_valid=is_valid)
        self.status = status
        self.data = data

response = client.get(url)

Grid Cell with Obstacles

from f_core.mixins.validatable_public import ValidatablePublic

class Cell(ValidatablePublic):
    def __init__(self, row: int, col: int):
        ValidatablePublic.__init__(self, is_valid=True)
        self.row = row
        self.col = col

# Create grid
grid = [[Cell(r, c) for c in range(5)] for r in range(5)]

# Add obstacles
grid[1][1].set_invalid()
grid[2][2].set_invalid()

# Pathfinding
for row in grid:
    for cell in row:
        if cell:  # Walkable?
            visit(cell)

Algorithm Solution

class Solution(ValidatablePublic):
    def __init__(self, result: any):
        # Start as invalid, verify later
        ValidatablePublic.__init__(self, is_valid=False)
        self.result = result

    def verify(self) -> bool:
        # Run verification logic
        if self._check_constraints():
            self.set_valid()
            return True
        return False

# Usage
solution = algorithm.solve(problem)
if solution.verify():
    print("Valid solution found")

Filtering Collections

# Using truthiness in comprehensions
cells = [Cell(0, 0), Cell(1, 1), Cell(2, 2)]
cells[1].set_invalid()

valid = [c for c in cells if c]      # [Cell(0,0), Cell(2,2)]
invalid = [c for c in cells if not c]  # [Cell(1,1)]

# Using filter()
valid = list(filter(bool, cells))

# Using any()/all()
has_valid = any(cells)    # True if any cell is valid
all_valid = all(cells)    # True if all cells are valid

Process Execution

class Process(HasName, Validatable):
    def __init__(self, name: str):
        HasName.__init__(self, name=name)
        Validatable.__init__(self)  # None = not run yet

    def run(self) -> bool:
        try:
            self._execute()
            self._is_valid = True
        except Exception:
            self._is_valid = False
        return self._is_valid

# Usage
process = Process("DataSync")
if not process:  # None → False (not run)
    process.run()
if process:  # Now True or False
    print("Success!")
