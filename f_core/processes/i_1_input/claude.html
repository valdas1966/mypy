What is ProcessInput?

A generic process class that accepts typed input data. Extends ProcessABC with an input property, enabling type-safe data flow into your process logic.

Generic Typing

ProcessInput[T] for type safety

Input Property

Read-only access to input data

Inherits Lifecycle

run(), elapsed from ProcessABC

Level 1 Process

Single capability: input handling

Quick Start

from f_core.processes.i_1_input import ProcessInput

class DataLoader(ProcessInput[str]):
    def _run(self) -> None:
        # Access typed input
        filepath = self.input
        self.load_file(filepath)

loader = DataLoader(input='/path/to/data.csv', name='CSV Loader')
loader.run()
print(f"Loaded from: {loader.input}")

Basic Subclass

from f_core.processes.i_1_input import ProcessInput

class DataLoader(ProcessInput[str]):
    def _run(self) -> None:
        # Access typed input
        filepath = self.input
        print(f"Loading: {filepath}")

loader = DataLoader(input='/data/file.csv', name='CSV Loader')
loader.run()

print(f"Process: {loader.name}")
print(f"Input: {loader.input}")
print(f"Elapsed: {loader.elapsed} seconds")

With Integer Input

class Calculator(ProcessInput[int]):
    def _run(self) -> None:
        result = self.input * 2
        print(f"Input {self.input} doubled = {result}")

calc = Calculator(input=5, name='Doubler')
calc.run()  # Output: Input 5 doubled = 10

With Complex Input

from dataclasses import dataclass

@dataclass
class Config:
    path: str
    batch_size: int

class ConfiguredProcess(ProcessInput[Config]):
    def _run(self) -> None:
        print(f"Path: {self.input.path}")
        print(f"Batch: {self.input.batch_size}")

config = Config(path='/data', batch_size=32)
process = ConfiguredProcess(input=config, name='Configured')
process.run()

Factory Usage

# Get the factory-created class
A = ProcessInput.Factory.a()

# Instantiate with input
proc = A(input=1)
proc.run()

print(proc.input)  # 1
print(proc.name)   # 'A'

Sorting Processes

# Processes are sortable by name (from HasName)
processes = [
    DataLoader(input='/z.csv', name='Zebra'),
    DataLoader(input='/a.csv', name='Alpha'),
    DataLoader(input='/b.csv', name='Beta')
]

sorted(processes)  # [Alpha, Beta, Zebra]
