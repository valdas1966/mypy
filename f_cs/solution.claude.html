Overview

SolutionAlgo is the generic base class for algorithm output. It contains validity status, statistics, and can be extended with algorithm-specific results.

class SolutionAlgo(Generic[Stats], ValidatablePublic, HasRecord):

Usage Examples

Custom Solution with Results

class SortingSolution(SolutionAlgo[SortingStats]):
    RECORD_SPEC = {
        'is_valid': lambda o: bool(o),
        'stats': lambda o: o.stats.record,
        'result': lambda o: o.result
    }

    def __init__(self,
                 is_valid: bool,
                 stats: SortingStats,
                 result: list[int]):
        super().__init__(is_valid=is_valid, stats=stats)
        self._result = result

    @property
    def result(self) -> list[int]:
        return self._result

Use in Algorithm

class SortingAlgo(Algo[SortingProblem, SortingSolution]):
    def _run(self) -> None:
        sorted_data = sorted(self.problem.data)

        self._output = SortingSolution(
            is_valid=True,
            stats=self._stats,
            result=sorted_data
        )

# Execute
algo = SortingAlgo(problem=SortingProblem([5, 2, 8]))
solution = algo.run()

if solution:
    print(f"Sorted: {solution.result}")
    print(f"Time: {solution.stats.elapsed}s")
