Overview

StatsAlgo is the base class for tracking algorithm metrics. It provides a standard way to collect and report statistics like execution time, and can be extended for algorithm-specific metrics.

class StatsAlgo:
    def __init__(self, elapsed: int = 0) -> None:
        self._elapsed = elapsed

    @property

Extending with Custom Metrics

class SortingStats(StatsAlgo):
    def __init__(self):
        super().__init__()
        self._comparisons = 0
        self._swaps = 0

    @property
    def comparisons(self) -> int:
        return self._comparisons

    @comparisons.setter
    def comparisons(self, value: int) -> None:
        self._comparisons = value

    @property
    def swaps(self) -> int:
        return self._swaps

    @swaps.setter
    def swaps(self, value: int) -> None:
        self._swaps = value

    @property
    def record(self) -> dict:
        return {
            'elapsed': self.elapsed,
            'comparisons': self.comparisons,
            'swaps': self.swaps
        }

Integration with Algo

The Algo class handles stats automatically:

class Algo(ProcessIO[Problem, Solution]):
    cls_stats: type[StatsAlgo] = StatsAlgo  # Override for custom stats

    def __init__(self, problem, name):
        super().__init__(input=problem, name=name)
        self._stats = self.cls_stats()  # Create stats instance

    def _run_post(self):
        super()._run_post()
        self._stats.elapsed = self.elapsed  # Auto-set elapsed
