# minimal usage
from f_overleaf import OverLeaf

overleaf = OverLeaf.Factory.valdas()
project  = overleaf['Test']

project.create_folder('figs')
project.create_file('figs/a.tex', r'\documentclass{article}')
project.upload_file('/tmp/main.tex', 'main.tex')
